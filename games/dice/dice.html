Qiskit Algorithm Reference

from qiskit import QuantumCircuit
from qiskit_aer import AerSimulator

qc = QuantumCircuit(3,3)

qc.h(0)
qc.h(1)
qc.h(2)

qc.measure([0,1,2],[0,1,2])

sim = AerSimulator()
result = sim.run(qc).result()

counts = result.get_counts()

print(counts)
